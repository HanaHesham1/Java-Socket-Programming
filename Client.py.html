import time, socket, sys

print("\nWelcome to Chat Room\n")
print("Initialising....\n")
time.sleep(1)

s = socket.socket()
shost = socket.gethostname()
ip = socket.gethostbyname(shost)
print(shost, "(", ip, ")\n")
port = 5555

time.sleep(1)
s.connect(("127.0.0.1", port))
print("Connected...\n")


s_name = s.recv(1024)
s_name = s_name.decode()
print(s_name, "has joined the chat room\nEnter close socket to exit chat room\n")

name = input(str("\nEnter the message you want to capitalise: "))
s.send(name.encode())
print(name.upper())
while True:
    message = s.recv(1024)
    message = message.decode()
    print(s_name, ":", message)
    message = input(str("Me : "))
    if message == "close socket":
        print("Left chat room!")
        s.send(message.encode())
        print("\n")
        s.close()
        break
    s.send(message.encode())
